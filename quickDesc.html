Quick sort is a divide-and-conquer algorithm that picks an element as a pivot and partitions the array around the pivot. It recursively sorts the sub-arrays formed by dividing the array into elements less than and greater than the pivot.




pseudocode
function swap(arr, xp, yp)
{
   var temp = arr[xp];
   arr[xp] = arr[yp];
    arr[yp] = temp;
}


function partition(arr, low, high)
{
    var pivot = arr[high];
    var i = (low - 1);
    for (var j = low; j <= high - 1; j++)
{
        if (arr[j] < pivot)
{
            i++;
            swap(arr, i, j);
}
}
